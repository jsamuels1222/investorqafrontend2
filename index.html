    <script>

        // Replace with your actual Render.com API endpoint

        const API_ENDPOINT = 'https://investorqa2.onrender.com';

       

        // Function to handle cold starts

        async function warmupAPI() {

            try {

                await fetch(API_ENDPOINT.replace('/answer', ''));

                console.log('API warmed up');

            } catch (error) {

                console.log('Warmup request sent');

            }

        }

       

        // Warm up API when page loads

        warmupAPI();

       

        document.getElementById('qaForm').addEventListener('submit', async (e) => {

            e.preventDefault();

           

            const company = document.getElementById('company').value;

            const question = document.getElementById('question').value;

            const submitButton = document.getElementById('submitButton');

            const loadingSpinner = document.getElementById('loadingSpinner');

            const answerContainer = document.getElementById('answerContainer');

            const errorMessage = document.getElementById('errorMessage');

           

            // Reset UI

            submitButton.disabled = true;

            loadingSpinner.classList.remove('hidden');

            answerContainer.classList.add('hidden');

            errorMessage.style.display = 'none';

           

            try {

                // Warm up API first

                await warmupAPI();

               

                // Wait a bit if it was cold

                await new Promise(resolve => setTimeout(resolve, 2000));

               

                const response = await fetch(API_ENDPOINT, {

                    method: 'POST',

                    headers: {

                        'Content-Type': 'application/json',

                    },

                    body: JSON.stringify({

                        query: question,

                        company_ticker: company

                    })

                });

               

                if (!response.ok) {

                    throw new Error(`HTTP error! status: ${response.status}`);

                }

               

                const data = await response.json();

               

                // Display the answer

                document.getElementById('answerText').textContent = data.response;

               

                // Display sources

                const sourcesList = document.getElementById('sourcesList');

                sourcesList.innerHTML = '';

                data.sources.forEach(source => {

                    const li = document.createElement('li');

                    li.textContent = source;

                    sourcesList.appendChild(li);

                });

               

                answerContainer.classList.remove('hidden');

            } catch (error) {

                console.error('Error:', error);

                errorMessage.textContent = 'Sorry, there was an error processing your request. Please try again in a moment.';

                errorMessage.style.display = 'block';

            } finally {

                submitButton.disabled = false;

                loadingSpinner.classList.add('hidden');

            }

        });

    </script>
